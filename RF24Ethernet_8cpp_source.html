119 {

124  uip_buf = (uint8_t*)&network.frag_ptr->message_buffer[0];

202 void RF24EthernetClass::tick()
203 {
204  #if defined (ARDUINO_ARCH_ESP8266)
205  yield();
206  #endif
207  if (RF24Ethernet.network.update() == EXTERNAL_DATA_TYPE) {
208  if (RF24Ethernet.network.frag_ptr->message_size <= UIP_BUFSIZE) {
209  uip_len = RF24Ethernet.network.frag_ptr->message_size;
210  }
211  }
212 
213  #if !defined (RF24_TAP)
214  if (uip_len > 0) {
215  uip_input();
216  if (uip_len > 0) {
217  network_send();
218  }
219  }
220  else if (timer_expired(&Ethernet.periodic_timer)) {
221  timer_reset(&Ethernet.periodic_timer);
222  for(int i = 0; i < UIP_CONNS; i++) {
223  uip_periodic(i);
224  /* If the above function invocation resulted in data that
225  should be sent out on the network, the global variable
226  uip_len is set to a value > 0. */
227  if (uip_len > 0) {
228  network_send();
229  }
230  }
231  }
232  #else // defined (RF24_TAP)
233  if (uip_len > 0) {
234  if (BUF->type == htons(UIP_ETHTYPE_IP)) {
235  uip_arp_ipin();
236  uip_input();
237  /* If the above function invocation resulted in data that
238  should be sent out on the network, the global variable
239  uip_len is set to a value > 0. */
240  if (uip_len > 0) {
241  uip_arp_out();
242  network_send();
243  }
244  }
245  else if (BUF->type == htons(UIP_ETHTYPE_ARP)) {
246  uip_arp_arpin();
247  /* If the above function invocation resulted in data that
248  should be sent out on the network, the global variable
249  uip_len is set to a value > 0. */
250  if (uip_len > 0) {
251  network_send();
252  }
253  }
254  }
255  else if (timer_expired(&Ethernet.periodic_timer)) {
256  timer_reset(&Ethernet.periodic_timer);
257  for(int i = 0; i < UIP_CONNS; i++) {
258  uip_periodic(i);
259  /* If the above function invocation resulted in data that
260  should be sent out on the network, the global variable
261  uip_len is set to a value > 0. */
262  if (uip_len > 0) {
263  uip_arp_out();
264  network_send();
265  }
266  }
267  #endif // defined (RF24_TAP)
268  #if UIP_UDP
269  for(int i = 0; i < UIP_UDP_CONNS; i++) {
270  uip_udp_periodic(i);
271  /* If the above function invocation resulted in data that
272  should be sent out on the network, the global variable
273  uip_len is set to a value > 0. */
274  if (uip_len > 0) {
275  //uip_arp_out();
276  //network_send();
277  RF24UDP::_send((uip_udp_userdata_t *)(uip_udp_conns[i].appstate));
278  }
279  }
280  #endif /* UIP_UDP */
281  #if defined (RF24_TAP)
282  /* Call the ARP timer function every 10 seconds. */
283 
284  if (timer_expired(&Ethernet.arp_timer)) {
285  timer_reset(&Ethernet.arp_timer);
286  uip_arp_timer();
287  }
288  }
289 #endif //RF24_TAP
290 }

295 {
296  RF24NetworkHeader headerOut(00,EXTERNAL_DATA_TYPE);
297  #if defined ETH_DEBUG_L1 || defined ETH_DEBUG_L2
298  bool ok = RF24Ethernet.network.write(headerOut, uip_buf, uip_len);
299  #else
300  RF24Ethernet.network.write(headerOut, uip_buf, uip_len);
301  #endif
302 
303  #if defined ETH_DEBUG_L1 || defined ETH_DEBUG_L2
304  if (!ok) {
305  Serial.println(); Serial.print(millis()); Serial.println(F(" *** RF24Ethernet Network Write Fail ***"));
306  }
307  #endif
308  #if defined ETH_DEBUG_L2
309  if (ok) {
310  Serial.println(); Serial.print(millis()); Serial.println(F(" RF24Ethernet Network Write OK"));
311  }
312  #endif
313 
314 }
